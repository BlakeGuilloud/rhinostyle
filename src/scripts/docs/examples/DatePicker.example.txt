

class ComponentExample extends React.Component {
  state = {
    startDate: null,
  };

  _handleChange = (date) => {
    this.setState({ startDate: date });
  };

  render() {
    return (
      <div className="row row--condensed">
        <div className="col-sm-2">
          <DatePicker className="form__control" selected={this.state.startDate} onChange={this._handleChange} placeholderText="Click to select a date" />
        </div>
      </div>
    );
  }
}

ReactDOM.render(<ComponentExample />, mountNode);

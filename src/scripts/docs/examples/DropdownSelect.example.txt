class ComponentExample extends React.Component {
  state = {
    activeKey: null,
  };

  updateActiveKey = (index) => {
    this.setState({
      activeKey: index,
    });
  };

  render() {
    return (
    <div>
      <DropdownSelect label="Dropdown Select" type="primary" activeKey={this.state.activeKey} onSelect={this.updateActiveKey}>
        <DropdownMenuHeader>Menu Header</DropdownMenuHeader>
        <DropdownMenuItem id={1}>Item</DropdownMenuItem>
        <DropdownMenuItem id={2}>Another Item</DropdownMenuItem>
        <DropdownMenuItem id={3}>A third item</DropdownMenuItem>
        <DropdownMenuDivider />
        <DropdownMenuItem id={4}>Separated item</DropdownMenuItem>
      </DropdownSelect>
    </div>
    );
  }
}

ReactDOM.render(<ComponentExample />, mountNode);

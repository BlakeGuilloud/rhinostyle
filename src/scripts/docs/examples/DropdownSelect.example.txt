class ComponentExample extends React.Component {
  state = {
    activeKey: null,
    activeFilterKey: null,
  };

  updateActiveKey = (index) => {
    this.setState({
      activeKey: index,
    });
  };

  updateActiveFilterKey = (index) => {
    this.setState({
      activeFilterKey: index,
    });
  };

  render() {
    return (
    <div>
      <DropdownSelect label="Dropdown Select" type="primary" activeKey={this.state.activeKey} select={this.updateActiveKey}>
        <DropdownMenuHeader>Menu Header</DropdownMenuHeader>
        <DropdownMenuItem id={1}>Item</DropdownMenuItem>
        <DropdownMenuItem id={2}>Another Item</DropdownMenuItem>
        <DropdownMenuItem id={3}>A third item</DropdownMenuItem>
        <DropdownMenuDivider />
        <DropdownMenuItem id={4}>Separated item</DropdownMenuItem>
      </DropdownSelect>
      &nbsp;&nbsp;
      <DropdownSelect wide label="Dropdown Select w/ Filter" type="secondary" activeKey={this.state.activeFilterKey} select={this.updateActiveFilterKey}>
        <DropdownSelectFilter placeholder="Filter Contact">
            <DropdownMenuItem id={1}>Item</DropdownMenuItem>
            <DropdownMenuItem id={2}>Another Item</DropdownMenuItem>
            <DropdownMenuItem id={3}>A third item</DropdownMenuItem>
            <DropdownMenuItem id={4}>A fourth item</DropdownMenuItem>
            <DropdownMenuItem id={5}>A fifth item</DropdownMenuItem>
            <DropdownMenuItem id={6}>A sixth item</DropdownMenuItem>
            <DropdownMenuItem id={7}>A seventh item</DropdownMenuItem>
            <DropdownMenuItem id={8}>An eighth item</DropdownMenuItem>
        </DropdownSelectFilter>
      </DropdownSelect>
    </div>
    );
  }
}

ReactDOM.render(<ComponentExample />, mountNode);
